## Сортиране в друг масив

```cpp
int main() {  
    int a[] = {2, 8, 4, 0, 6, 9, 1, 4, 7};  
    int n = sizeof(a) / sizeof(int);  
    int b[n];  
  
    int max = a[0];  
    for (int i = 1; i < n; i++) {  
        if (a[i] > max) {  
            max = a[i];  
        }  
    }  
  
    b[n - 1] = max;  
  
    for (int i = 0; i < n - 1; i++) {  
        int min = a[0];  
        int flag = 0;  
        for (int j = 1; j < n; j++) {  
            if (min > a[j]) {  
                min = a[j];  
                flag = j;  
            }  
        }  
        b[i] = min;  
        a[flag] = max;  
    }  
      
    return 0;  
}
```

## Сортиране с пряка селекция / Selection sort

```cpp
int main() {  
    int a[] = {3, 4, 9, 0, 4, 8};  
    int n = sizeof(a) / sizeof(int);  
  
    for (int i = 0; i < n - 1; i++) {  
        int min = a[i];  
        int flag = i;  
        for (int j = i + 1; j < n; j++) {  
            if (a[j] < min) {  
                min = a[j];  
                flag = j;  
            }  
        }  
        a[flag] = a[i];  
        a[i] = min;  
    }  
}
```
