–ù–∞–º–∏—Ä–∞ —Å–µ –Ω–∞–π-–≥–æ–ª—è–º–∞—Ç–∞ —Å—Ç–æ–π–Ω–æ—Å—Ç –Ω–∞ –≤—Ö–æ–¥–Ω–∏—è –º–∞—Å–∏–≤, –∑–∞–ø–∞–∑–≤–∞ —Å–µ –≤ –ø—Ä–æ–º–µ–Ω–ª–∏–≤–∞ –∏ —Å–µ –ø–æ—Å—Ç–∞–≤—è –Ω–∞ –ø–æ—Å–ª–µ–¥–Ω–æ –º—è—Å—Ç–æ –≤ "–¥—Ä—É–≥–∏—è –º–∞—Å–∏–≤" (—Ä–µ–∑—É–ª—Ç–∞—Ç–Ω–∏—è). –û–±—Ö–æ–∂–¥–∞–º–µ –≤—Ö–æ–¥–Ω–∏—è –º–∞—Å–∏–≤, –Ω–∞–º–∏—Ä–∞–º–µ –º–∏–Ω–∏–º–∞–ª–Ω–∞—Ç–∞ —Å—Ç–æ–π–Ω–æ—Å—Ç –∏ —è –ø–æ—Å—Ç–∞–≤—è–º–µ –Ω–∞ –Ω–∞–π-–º–∞–ª–∫–∞—Ç–∞ –Ω–µ–∑–∞–µ—Ç–∞ –ø–æ–∑–∏—Ü–∏—è –≤ —Ä–µ–∑—É–ª—Ç–∞—Ç–Ω–∏—è –º–∞—Å–∏–≤. –ú–∏–Ω–∏–º–∞–ª–Ω–∞—Ç–∞ —Å—Ç–æ–π–Ω–æ—Å—Ç –æ—Ç –≤—Ö–æ–¥–Ω–∏—è –º–∞—Å–∏–≤ –±–∏–≤–∞ –∑–∞–º–µ–Ω–µ–Ω–∞ —Å –Ω–∞–π-–≥–æ–ª—è–º–∞—Ç–∞ —Å—Ç–æ–π–Ω–æ—Å—Ç - –¥–µ–∑–∞–∫—Ç–∏–≤–∏—Ä–∞–º–µ.

### –ú–µ—Ç–æ–¥ –Ω–∞ "–ü—Ä—è–∫–∞ —Å–µ–ª–µ–∫—Ü–∏—è"

–û–±—Ö–æ–∂–¥–∞ —Å–µ –º–∞—Å–∏–≤–∞ –∏ –∑–∞–¥–∞–≤–∞–º–µ, —á–µ —Ç–µ–∫—É—â–∏—è –µ–ª–µ–º–µ–Ω—Ç (–∏ –Ω–µ–≥–æ–≤–∏—è –∏–Ω–¥–µ–∫—Å) –µ –Ω–∞–π-–º–∞–ª–∫–æ—Ç–æ —á–∏—Å–ª–æ. –í—ä–≤ –≤–ª–æ–∂–µ–Ω —Ü–∏–∫—ä–ª, –æ–±—Ö–æ–∂–¥–∞–º–µ "—Å–ª–µ–¥–≤–∞—â–∏—Ç–µ" –µ–ª–µ–º–µ–Ω—Ç–∏ —Å–ª–µ–¥ —Ç–µ–∫—É—â–∏—è –µ–ª–µ–º–µ–Ω—Ç (–Ω–∞ –≤—ä–Ω—à–Ω–∏—è —Ü–∏–∫—ä–ª), –∞–∫–æ –µ–ª–µ–º–µ–Ω—Ç –æ—Ç –≤–ª–æ–∂–µ–Ω–∏—è —Ü–∏–∫—ä–ª –±—ä–¥–µ –Ω–∞–º–µ—Ä–µ–Ω –∫–∞—Ç–æ –ø–æ-–º–∞–ª—ä–∫ –æ—Ç –≤–µ—á–µ –∑–∞–¥–∞–¥–µ–Ω–∏—è (–≤—ä–≤ –≤—ä–Ω—à–Ω–∏—è —Ü–∏–∫—ä–ª), –∑–∞–º–µ–Ω—è–º–µ –ø—Ä–æ–º–µ–Ω–ª–∏–≤–∏—Ç–µ, –∫–æ–∏—Ç–æ –ø–∞–∑—è—Ç –∏–Ω–¥–µ–∫—Å–∞ –∏ —Å—Ç–æ–π–Ω–æ—Å—Ç—Ç–∞ –Ω–∞ –Ω–∞–π-–º–∞–ª–∫–∏—è –µ–ª–µ–º–µ–Ω—Ç.

### –ú–µ—Ç–æ–¥ –Ω–∞ "–ü—Ä—è–∫–∞ —Ä–∞–∑–º—è–Ω–∞" / "–ú–µ—Ö—É—Ä—á–µ—Ç–æ"

–í—ä–Ω—à–Ω–∏—è—Ç —Ü–∏–∫—ä–ª –µ —Ü–∏–∫—ä–ª —Å—ä—Å —Å–ª–µ–¥-—É—Å–ª–æ–≤–∏–µ –∏ —Å–µ –∏–∑–ø—ä–ª–Ω—è–≤–∞, –¥–æ–∫–∞—Ç–æ –≤—ä–≤ –≤—ä—Ç—Ä–µ—à–Ω–∏—è—Ç —Ü–∏–∫—ä–ª –Ω—è–º–∞ –ø–æ–≤–µ—á–µ –∏–Ω–≤–µ—Ä—Å–∏–∏. –ö–∞—Ç–æ —Å–µ –Ω–∞–º–µ—Ä–∏ –∏–Ω–≤–µ—Ä—Å–∏—è –º–µ–∂–¥—É —Å—ä—Å–µ–¥–Ω–∏ –µ–ª–µ–º–µ–Ω—Ç–∏, —Ä–∞–∑–º–µ–Ω—è–º–µ –¥–≤–∞—Ç–∞ –µ–ª–µ–º–µ–Ω—Ç–∞. –í—ä—Ç—Ä–µ—à–Ω–∏—è —Ü–∏–∫—ä–ª —Å–µ –∏–∑–ø—ä–ª–Ω—è–≤–∞ –¥–æ `–¥—ä–ª–∂–∏–Ω–∞—Ç–∞ –Ω–∞ –µ–ª–µ–º–µ–Ω—Ç–∏—Ç–µ - –±—Ä–æ—è—á - 1`, –∑–∞ –¥–∞ —Å–µ –∏–∑–±–µ–≥–Ω–µ –æ–±—Ö–æ–∂–¥–∞–Ω–µ –Ω–∞ –≤–µ—á–µ "–Ω–∞—Ä–µ–¥–µ–Ω–∞—Ç–∞ —á–∞—Å—Ç".

–ê–ª–≥–æ—Ä–∏—Ç—ä–º—ä—Ç –Ω–∞—Ä–µ–∂–¥–∞ –º–∞—Å–∏–≤–∞ "–æ—Ç–∑–∞–¥-–Ω–∞–ø—Ä–µ–¥", —Ç.–µ. —á–∞—Å—Ç—Ç–∞ —Å "–Ω–∞–π-–≥–æ–ª–µ–º–∏" –µ–ª–µ–º–µ–Ω—Ç–∏ –ø—ä—Ä–≤–æ (–≤ –Ω–∏–∑—Ö–æ–¥—è—â —Ä–µ–¥). –ê–∫–æ –º–∞–∫—Å–∏–º–∞–ª–Ω–∏—è –µ–ª–µ–º–µ–Ω—Ç –µ –≤ –Ω–∞—á–∞–ª–æ—Ç–æ –Ω–∞ –º–∞—Å–∏–≤–∞, —Ç–æ–π —â–µ –±—ä–¥–µ "–∏–∑–±—É—Ç–∞–Ω" (—Å–º–µ–Ω–µ–Ω–∞ –ø–æ–∑–∏—Ü–∏—è—Ç–∞ —Å –≤—Å–µ–∫–∏ —Å–ª–µ–¥–≤–∞—â –µ–ª–µ–º–µ–Ω—Ç –æ—Ç –∏—Ç–µ—Ä–∞—Ü–∏—è—Ç–∞) –æ—Ç –≤—ä—Ç—Ä–µ—à–Ω–∏—è —Ü–∏–∫—ä–ª "–Ω–∞–≥–æ—Ä–µ" (–∫—ä–º –∫—Ä–∞—è –Ω–∞ –º–∞—Å–∏–≤–∞) –∫–∞—Ç–æ –º–µ—Ö—É—Ä—á–µ :)

**–°–ª–æ–∂–Ω–æ—Å—Ç**:

**–ë—Ä–æ–π —Å—Ä–∞–≤–Ω–µ–Ω–∏—è**: 
- –ù–∞–π-–ª–æ—à —Å–ª—É—á–∞–π: $\frac{n(n-1)}{2}$
- –ù–∞–π-–¥–æ–±—ä—Ä —Å–ª—É—á–∞–π: $n-1$


### –ú–µ—Ç–æ–¥ –Ω–∞ "–ü—Ä—è–∫–æ –≤–º—ä–∫–≤–∞–Ω–µ"
*Insertion sort*

–í—ä–Ω—à–µ–Ω —Ü–∏–∫—ä–ª —Å –±—Ä–æ—è, –∫–æ–π—Ç–æ –æ–±—Ö–æ–∂–¥–∞ –≤—Å–∏—á–∫–∏ –µ–ª–µ–º–µ–Ω—Ç–∏ –Ω–∞ –¥–∞–¥–µ–Ω –º–∞—Å–∏–≤. –í—ä—Ç—Ä–µ—à–Ω–∏—è —Ü–∏–∫—ä–ª —Å —É—Å–ª–æ–≤–∏–µ —Å–µ –∏–∑–ø—ä–ª–Ω—è–≤–∞, –¥–æ–∫–∞—Ç–æ –ø—Ä–µ–¥—Ö–æ–¥–Ω–∏—Ç–µ –µ–ª–µ–º–µ–Ω—Ç–∏ (—Å–ø—Ä—è–º–æ —Ç–µ–∫—É—â–∏—è –µ–ª–µ–º–µ–Ω—Ç –Ω–∞ –≤—ä–Ω—à–Ω–∏—è —Ü–∏–∫—ä–ª) –∏–º–∞—Ç –∏–Ω–≤–µ—Ä—Å–∏—è –≤ —Å—ä—Å–µ–¥—Å—Ç–≤–æ `a[j-1] > a[j]`. –ê–∫–æ —Å—ä—Å–µ–¥–Ω–∏—Ç–µ –µ–ª–µ–º–µ–Ω—Ç–∏ –∏–º–∞—Ç –∏–Ω–≤–µ—Ä—Å–∏—è, —Ç–µ—Ö–Ω–∏—Ç–µ —Å—Ç–æ–π–Ω–æ—Å—Ç–∏ —Å–µ —Ä–∞–∑–º–µ–Ω—è—Ç –∏ –ø—Ä–æ–¥—ä–ª–∂–≤–∞ –∏–∑–ø—ä–ª–Ω–µ–Ω–∏–µ—Ç–æ –Ω–∞ –≤—ä—Ç—Ä–µ—à–Ω–∏—è —Ü–∏–∫—ä–ª "–Ω–∞–¥–æ–ª—É" (–∫—ä–º –Ω–∞—á–∞–ª–æ—Ç–æ –Ω–∞ –º–∞—Å–∏–≤–∞).

**–°–ª–æ–∂–Ω–æ—Å—Ç**: 
- –ù–∞–π-–ª–æ—à —Å–ª—É—á–∞–π: $O(n^2)$ 
- –ù–∞–π-–¥–æ–±—ä—Ä —Å–ª—É—á–∞–π: $O(n)$

**–ë—Ä–æ–π —Å—Ä–∞–≤–Ω–µ–Ω–∏—è**: 
- –ù–∞–π-–ª–æ—à —Å–ª—É—á–∞–π: $\frac{n(n-1)}{2}$
- –ù–∞–π-–¥–æ–±—ä—Ä —Å–ª—É—á–∞–π: $n-1$

## –ú–∞—Ç—Ä–∏—Ü–∏

### –£–º–Ω–æ–∂–µ–Ω–∏–µ

–ü—Ä–∏ —É–º–Ω–æ–∂–µ–Ω–∏–µ –Ω–∞ –º–∞—Ç—Ä–∏—Ü–∏ $A (m \times n)$ –∏ $B (n \times p)$:
- —Å–µ —É–º–Ω–æ–∂–∞–≤–∞ —Ä–µ–¥ –æ—Ç $A$ –ø–æ –∫–æ–ª–æ–Ω–∞ –æ—Ç $B$
- –æ–ø–µ—Ä–∞—Ü–∏—è—Ç–∞ –Ω–µ –µ –∫–æ–º—É—Ç–∞—Ç–∏–≤–Ω–∞ (–Ω–µ –º–æ–∂–µ –¥–∞ —Å–µ —Ä–∞–∑–º–µ–Ω—è—Ç –º–µ—Å—Ç–∞—Ç–∞ –Ω–∞ –º–∞—Ç—Ä–∏—Ü–∏—Ç–µ)
- –∏–∑—Ö–æ–¥–Ω–∞—Ç–∞ –º–∞—Ç—Ä–∏—Ü–∞ –µ —Å —Ä–∞–∑–º–µ—Ä $m \times p$
- –±—Ä–æ—è —Ä–µ–¥–æ–≤–µ –Ω–∞ $A$ —Ç—Ä—è–±–≤–∞ –¥–∞ –æ—Ç–≥–æ–≤–∞—Ä—è –Ω–∞ –±—Ä–æ—è –∫–æ–ª–æ–Ω–∏ –Ω–∞ $B$
- –±—Ä–æ—è —É–º–Ω–æ–∂–µ–Ω–∏—è –µ $m \cdot n \cdot p$ 
- –±—Ä–æ—è —Å—ä–±–∏—Ä–∞–Ω–∏—è –µ $m \cdot (n-1) \cdot p$

**–°–ª–æ–∂–Ω–æ—Å—Ç**: $O(n^3)$ (–∫–≤–∞–¥—Ä–∞—Ç–Ω–∞ –º–∞—Ç—Ä–∏—Ü–∞)

**–ë—Ä–æ–π –æ–ø–µ—Ä–∞—Ü–∏–∏**: 
- –£–º–Ω–æ–∂–µ–Ω–∏—è: $n^3$

int main() {  
    int n = 3;  
    float a[3][3] = {  
            {-3.0, -1.0, 2.0},  
            {2.0,  1.0,  -1.0},  
            {-2.0, 1.0,  2.0}  
    };  
    float b[] = {-11.0, 8.0, -3.0};  
    
	// function that just prints the matrix's rows & lines
    print_matrix(a, 3);
  
    for (int i = 0; i < n; ++i) {  
	    // element on the diagonal  
        float first_element = a[i][i]; 
        b[i] /= first_element;  
        for (int j = i; j < n; ++j) {  
            a[i][j] /= first_element;  
        }  
  
        // for the next rows (below)  
        for (int k = i + 1; k < n; ++k) {  
            // how many times this row needs to be subtracted  
            // from the current (i) row to zero its elements            
            float row_coefficient = a[k][i];  
            // only elements after the diagonal element  
            // of the current row            
            for (int l = i; l < n; ++l) {  
                a[k][l] = a[k][l] - row_coefficient * a[i][l];  
            }  
            b[k] = b[k] - row_coefficient * b[i];  
        }  
    }  
  
    std::cout << std::endl;  
    print_matrix(a, n);  
}
